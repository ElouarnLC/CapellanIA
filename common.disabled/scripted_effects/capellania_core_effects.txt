# CapellanIA: Core Scripted Effects for Data Engine
# Target Scope: Character (Sovereign)

capellania_initialize_character_effect = {
	if = {
		limit = {
			not = { has_character_flag = chapel_is_active }
		}
		add_character_flag = chapel_is_active
		
		# Set baseline status to resident
		add_character_flag = chapel_status_resident
		
		# Initialize core numerical tracking variables (0-100 scale)
		set_variable = { name = chapel_prestige value = 50 }
		set_variable = { name = chapel_liturgical_capacity value = 10 }
		set_variable = { name = chapel_liturgical_load value = 0 }
		set_variable = { name = chapel_discipline value = 80 }
		set_variable = { name = chapel_conformity value = 80 }
		
		# Initialize slot upgrade levels (0 = None/Empty)
		set_variable = { name = chapel_slot_autel value = 0 }
		set_variable = { name = chapel_slot_vitraux value = 0 }
		set_variable = { name = chapel_slot_tombeaux value = 0 }
		set_variable = { name = chapel_slot_mobilier value = 0 }
	}
}

capellania_set_status_resident_effect = {
	remove_character_flag = chapel_status_travel_light
	remove_character_flag = chapel_status_travel_ordinary
	remove_character_flag = chapel_status_travel_solemn
	add_character_flag = chapel_status_resident
}

capellania_set_status_travel_light_effect = {
	remove_character_flag = chapel_status_resident
	remove_character_flag = chapel_status_travel_ordinary
	remove_character_flag = chapel_status_travel_solemn
	add_character_flag = chapel_status_travel_light
}

capellania_set_status_travel_ordinary_effect = {
	remove_character_flag = chapel_status_resident
	remove_character_flag = chapel_status_travel_light
	remove_character_flag = chapel_status_travel_solemn
	add_character_flag = chapel_status_travel_ordinary
}

capellania_set_status_travel_solemn_effect = {
	remove_character_flag = chapel_status_resident
	remove_character_flag = chapel_status_travel_light
	remove_character_flag = chapel_status_travel_ordinary
	add_character_flag = chapel_status_travel_solemn
}

capellania_deactivate_effect = {
	remove_character_flag = chapel_is_active
	remove_character_flag = chapel_status_resident
	remove_character_flag = chapel_status_travel_light
	remove_character_flag = chapel_status_travel_ordinary
	remove_character_flag = chapel_status_travel_solemn
	
	clear_variable = chapel_prestige
	clear_variable = chapel_liturgical_capacity
	clear_variable = chapel_liturgical_load
	clear_variable = chapel_discipline
	clear_variable = chapel_conformity
	
	clear_variable = chapel_slot_autel
	clear_variable = chapel_slot_vitraux
	clear_variable = chapel_slot_tombeaux
	clear_variable = chapel_slot_mobilier
	clear_variable = chapel_dedication
	clear_variable = chapel_dotation
	clear_variable = chapel_canon_status
}
