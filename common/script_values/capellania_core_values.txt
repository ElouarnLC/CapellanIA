# --- Liturgical Capacity engine ---
# Capacity = staff + books + furniture + endowment, weakened while the court travels.

capellania_books_capacity_bonus = {
	value = capellania_books_value
	multiply = 2
}

capellania_capacity_total = {
	value = 4
	if = {
		limit = { employs_court_position = capellania_premier_chapelain }
		add = 4
	}
	if = {
		limit = { employs_court_position = capellania_maitre_de_chapelle }
		add = 2
	}
	if = {
		limit = { employs_court_position = capellania_chantre_principal }
		add = 1
	}
	if = {
		limit = { employs_court_position = capellania_gardien_du_tresor }
		add = 1
	}
	add = capellania_books_capacity_bonus
	add = capellania_slot_mobilier_value
	if = {
		limit = {
			has_variable = chapel_dotation
			var:chapel_dotation >= 2
		}
		add = 1
	}
	if = {
		limit = {
			has_variable = chapel_dotation
			var:chapel_dotation >= 3
		}
		add = 1
	}
	# Residence Stability: staff travelling with the sovereign weakens the seat
	if = {
		limit = {
			has_variable = chapel_travel_status
			var:chapel_travel_status = 2
		}
		subtract = 2
	}
	if = {
		limit = {
			has_variable = chapel_travel_status
			var:chapel_travel_status = 3
		}
		subtract = 4
	}
	min = 1
}

# Load = daily offices (2) + 2 per founded anniversary mass
capellania_load_total = {
	value = capellania_founded_masses_value
	multiply = 2
	add = 2
}

capellania_overload_value = {
	value = capellania_load_total
	subtract = capellania_capacity_total
	min = 0
}

# Fictive Masses ticker phase: 0 = none, 1 = piety decay, 2 = bishop hook, 3 = public scandal
capellania_debt_phase = {
	value = 0
	if = {
		limit = {
			has_variable = chapel_liturgical_debt
			var:chapel_liturgical_debt >= 1
		}
		value = 1
	}
	if = {
		limit = {
			has_variable = chapel_liturgical_debt
			var:chapel_liturgical_debt >= 3
		}
		value = 2
	}
	if = {
		limit = {
			has_variable = chapel_liturgical_debt
			var:chapel_liturgical_debt >= 5
		}
		value = 3
	}
}
