# CapellanIA: Scripted GUIs for UI and Logic Interface
# Target Scope: Character (Sovereign)

capellania_open_window = {
	scope = character
	is_shown = {
		has_character_flag = chapel_window_opened
	}
	effect = {
		set_character_flag = chapel_window_opened
	}
}

capellania_close_window = {
	scope = character
	effect = {
		remove_character_flag = chapel_window_opened
	}
}

# 1. AUTEL PURCHASE LOGIC (Flemish Sculpted Retable)
capellania_purchase_retable = {
	scope = character
	is_shown = {
		has_character_flag = chapel_is_active
		not = {
			has_variable = chapel_slot_autel
			compare_value = {
				value = character_variable:chapel_slot_autel
				op = >=
				compare_to = 2
			}
		}
	}
	is_valid = {
		gold >= 145
	}
	effect = {
		pay_short_term_gold_cost = 145
		
		# Historical uncertainty roll: 85% success, 15% failure
		random = {
			chance = 85
			change_variable = { name = chapel_prestige change = 15 }
			set_variable = { name = chapel_slot_autel value = 2 }
			send_interface_toast = {
				title = capellania.retable_success.title
				desc = capellania.retable_success.desc
			}
		}
		random = {
			chance = 15
			change_variable = { name = chapel_prestige change = -5 }
			set_variable = { name = chapel_slot_autel value = 1 }
			send_interface_toast = {
				title = capellania.retable_fail.title
				desc = capellania.retable_fail.desc
			}
		}
	}
}
