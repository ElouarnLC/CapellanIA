# CapellanIA: Travel Options Setup

capellania_travel_light = {
	can_select = {
		has_character_flag = chapel_is_active
		is_adult = yes
	}
	
	cost = {
		gold = 5
	}
	
	on_select = {
		capellania_set_status_travel_light_effect = yes
	}
	
	character_modifier = {
		piety_gain_mult = 0.02
	}
}
