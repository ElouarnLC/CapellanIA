# CapellanIA: Interactive Foundation Event Chain
namespace = capellania_found_events

# Event 1: Dedication Setup
capellania_found_events.0001 = {
	type = character_event
	title = capellania_found_events.0001.t
	desc = capellania_found_events.0001.d
	theme = religion
	
	left_portrait = root
	
	option = {
		name = capellania_found_events.0001.opt_notre_dame
		set_variable = { name = chapel_dedication value = 1 } # Notre-Dame
		trigger_event = { id = capellania_found_events.0002 }
	}
	option = {
		name = capellania_found_events.0001.opt_saint_michel
		set_variable = { name = chapel_dedication value = 2 } # Saint-Michel (military/dynastic)
		trigger_event = { id = capellania_found_events.0002 }
	}
	option = {
		name = capellania_found_events.0001.opt_saint_martin
		set_variable = { name = chapel_dedication value = 3 } # Saint-Martin (monastic/charity)
		trigger_event = { id = capellania_found_events.0002 }
	}
}

# Event 3: Canonical Authorization Setup
capellania_found_events.0003 = {
	type = character_event
	title = capellania_found_events.0003.t
	desc = capellania_found_events.0003.d
	theme = religion
	
	left_portrait = root
	
	option = {
		name = capellania_found_events.0003.opt_regular
		set_variable = { name = chapel_canon_status value = 1 } # Canonical approval
		
		# Execute core initialization
		capellania_initialize_character_effect = yes
		
		# Boost initial conformity due to Bishop's license
		change_variable = { name = chapel_conformity change = 15 }
		
		send_interface_toast = {
			title = capellania.found_success.title
			desc = capellania.found_success.desc
		}
	}
	option = {
		name = capellania_found_events.0003.opt_clandestine
		set_variable = { name = chapel_canon_status value = 2 } # Clandestine construction
		
		capellania_initialize_character_effect = yes
		
		# Suffer conformity penalty for illegal structure
		change_variable = { name = chapel_conformity change = -35 }
		
		send_interface_toast = {
			title = capellania.found_clandestine.title
			desc = capellania.found_clandestine.desc
		}
	}
}
